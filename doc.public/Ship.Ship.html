Module type Ship.Ship

module type Ship = sig .. end
type t

t is the type of ships

val taken : Ship.coor list Stdlib.ref

taken is the list of coordinates being used.

val empty : t

empty is an empty ship.

val is_empty : t -> bool

is_empty s is true iff s is empty.

val size : t -> int

size s is number of coordinates in s. size empty is 0.

val insert : Ship.coor -> t -> t

insert coor s inserts coordinate coor into ship s. Does nothing if coor already in s.

val remove : Ship.coor -> t -> t

remove coor s removes the coordinate coor from ship s. Does nothing if coor already in s.

val create : Ship.coor list -> t

create lst creates a ship with the coordinate list lst

val hit : Ship.coor -> t -> t

hit coord s is the ship s with the coordinate coord set to false.

val alive : t -> bool

alive s is true if there exists a coordinate in ship s that is true.

val compare : t -> t -> Ship.compare

compare s1 s2 returns EQ if and only if the first coordinates of s1 and s2 are equal, else returns GL.
